package main

import (
	"crypto/md5"
	"fmt"
	"net/http"
)

// hashPassword hashes a password with a broken algorithm (CWE-327, weak crypto).
func hashPassword(w http.ResponseWriter, r *http.Request) {
	pw := r.URL.Query().Get("password")
	sum := md5.Sum([]byte(pw))
	fmt.Fprintf(w, "%x\n", sum)
}

func main() {
	http.HandleFunc("/hash", hashPassword)
	http.ListenAndServe(":8080", nil)
}
